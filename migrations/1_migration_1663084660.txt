 CREATE TABLE IF NOT EXISTS author 
                    (id SERIAL PRIMARY KEY,
                    name VARCHAR (255) , surname VARCHAR (255) );
##############
 CREATE TABLE IF NOT EXISTS book 
                    (id SERIAL PRIMARY KEY,
                    title VARCHAR (255) , author INTEGER REFERENCES author (id) ON DELETE CASCADE );
##############
 CREATE TABLE IF NOT EXISTS course 
                    (id SERIAL PRIMARY KEY,
                    title VARCHAR (255) );
##############
 CREATE TABLE IF NOT EXISTS student_course 
                    (id INTEGER PRIMARY KEY, 
                    student_id INTEGER REFERENCES student (id) ON DELETE CASCADE, 
                    course_id INTEGER REFERENCES course (id) ON DELETE CASCADE) 
##############
 CREATE TABLE IF NOT EXISTS passportowner 
                    (id SERIAL PRIMARY KEY,
                    name VARCHAR (50) );
##############
 CREATE TABLE IF NOT EXISTS passport 
                    (id SERIAL PRIMARY KEY,
                    owner INTEGER UNIQUE REFERENCES passportowner (id) ON DELETE CASCADE , number VARCHAR (50) );
##############
 CREATE TABLE IF NOT EXISTS person 
                    (id SERIAL PRIMARY KEY,
                    name VARCHAR (50)  DEFAULT 'Bob' , surname VARCHAR (50)  NULL, test VARCHAR (50) , age  INTEGER );
##############
 CREATE TABLE IF NOT EXISTS student 
                    (id SERIAL PRIMARY KEY,
                    name VARCHAR (255) );
